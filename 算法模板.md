### 二分查找法

```python
from typing import List

class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        n = len(nums)
        left = 0
        right = n # 我们定义target在左闭右开的区间里，[left, right)  
        while left < right: # 因为left == right的时候，在[left, right)是无效的空间
            middle = left + ((right - left) >> 1)
            if nums[middle] > target:
                right = middle # target 在左区间，因为是左闭右开的区间，nums[middle]一定不是我们的目标值，所以right = middle，在[left, middle)中继续寻找目标值
            elif nums[middle] < target:
                left = middle + 1 # target 在右区间，在 [middle+1, right)中
            else: # nums[middle] == target
                return middle # 数组中找到目标值的情况，直接返回下标
        return right
```

```python
from collections import deque

def levelOrder(root: TreeNode) -> List[List[int]]:
    result = []
    if not root:
        return result
    que = deque([root])
    
    while que:
        size = len(que)
        vec = []
        for _ in range(size): # 这里一定要使用固定大小size，不要使用 len(que)
            node = que.popleft()
            vec.append(node.val)   # 节点处理的逻辑
            if node.left:
                que.append(node.left)
            if node.right:
                que.append(node.right)
        result.append(vec)
    return result
```

### 并查集

```python
n = 1005 # 根据题意而定 
father = [0] * n

# 并查集初始化
def init():
    for i in range(n):
        father[i] = i

# 并查集里寻根的过程
def find(u: int) -> int:
    if u == father[u]:
        return u
    father[u] = find(father[u]) # 路径压缩
    return father[u]

# 将v->u 这条边加入并查集
def join(u: int, v: int) -> None:
    u = find(u)
    v = find(v)
    if u == v:
        return 
    father[v] = u

# 判断 u 和 v是否找到同一个根
def same(u: int, v: int) -> bool:
    u = find(u)
    v = find(v)
    return u == v
```

## 动态规划模板题

动态规划解题五步曲：

1. 确定 dp 数组及其下标的含义
2. 确定递推公式（状态转移方程）
3. dp 数组如何初始化
4. 确定遍历顺序
5. 举例推导 dp 数组（打印 debug）

### 1. 基础题目 (Basic DP)

基础 DP 通常是一维或者简单的二维网格，重点在于找准递推关系。

#### 一维基础 DP (例：509.斐波那契数、70.爬楼梯)

```python
def climbStairs(n: int) -> int:
    if n <= 2:
        return n
    
    # 1. dp[i]：爬到第 i 阶有 dp[i] 种方法
    dp = [0] * (n + 1)
    
    # 2. 初始化
    dp[1] = 1
    dp[2] = 2
    
    # 3. 遍历顺序：从前向后
    for i in range(3, n + 1):
        # 4. 递推公式
        dp[i] = dp[i - 1] + dp[i - 2]
        
    return dp[n]

# 状态压缩版 (O(1) 空间)
def climbStairs_compressed(n: int) -> int:
    if n <= 2: return n
    a, b = 1, 2
    for _ in range(3, n + 1):
        a, b = b, a + b
    return b
```

#### 二维网格 DP (例：62.不同路径)

```python
def uniquePaths(m: int, n: int) -> int:
    # 1. dp[i][j]：走到 (i, j) 的路径数
    dp = [[0] * n for _ in range(m)]
    
    # 2. 初始化第一行和第一列
    for i in range(m): dp[i][0] = 1
    for j in range(n): dp[0][j] = 1
        
    # 3. 遍历顺序：从左到右，从上到下
    for i in range(1, m):
        for j in range(1, n):
            # 4. 递推公式
            dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
            
    return dp[m - 1][n - 1]
```

### 2. 背包问题 (Knapsack DP)

#### 01 背包 (每个物品只能用一次)

**场景：** 物品数量为1，求最大价值或能否装满（例：416.分割等和子集）。

```python
def knapsack_01(weight: list[int], value: list[int], bag_weight: int) -> int:
    # 1. dp[j]：容量为 j 的背包能装下的最大价值
    dp = [0] * (bag_weight + 1)
    
    # 2. 先遍历物品，再遍历背包
    for i in range(len(weight)):
        # 3. 01背包核心：背包容量一定要【倒序遍历】，防止物品被重复放入
        for j in range(bag_weight, weight[i] - 1, -1):
            dp[j] = max(dp[j], dp[j - weight[i]] + value[i])
            
    return dp[bag_weight]
```

#### 完全背包 (每个物品有无数个)

**场景：** 物品数量无限，求最大价值、装满方法数（例：322.零钱兑换、518.零钱兑换II）。

* **组合问题**（不强调顺序）：先遍历物品，再遍历背包。
* **排列问题**（强调顺序）：先遍历背包，再遍历物品。

```python
def knapsack_complete(weight: list[int], value: list[int], bag_weight: int) -> int:
    dp = [0] * (bag_weight + 1)
    
    # 先遍历物品，再遍历背包（适用于求组合数或最大价值）
    for i in range(len(weight)):
        # 完全背包核心：背包容量【正序遍历】，因为物品可以重复放入
        for j in range(weight[i], bag_weight + 1):
            dp[j] = max(dp[j], dp[j - weight[i]] + value[i])
            
    return dp[bag_weight]
```

### 3. 打家劫舍 (House Robber)

核心逻辑：相邻元素不能同时取。

#### 线性打家劫舍 (198. 打家劫舍)

```python
def rob(nums: list[int]) -> int:
    if not nums: return 0
    if len(nums) == 1: return nums[0]
    
    # dp[i]：考虑前 i 个房间，所能偷到的最大金额
    dp = [0] * len(nums)
    dp[0] = nums[0]
    dp[1] = max(nums[0], nums[1])
    
    for i in range(2, len(nums)):
        # 决定是偷当前房间（加上 dp[i-2]），还是不偷（保持 dp[i-1]）
        dp[i] = max(dp[i - 1], dp[i - 2] + nums[i])
        
    return dp[-1]
```

#### 树形打家劫舍 (337. 打家劫舍 III)

结合了二叉树的后序遍历。

```python
def robTree(root: TreeNode) -> int:
    # 返回一个长度为2的数组：[不偷当前节点的最大钱数, 偷当前节点的最大钱数]
    def traversal(node: TreeNode) -> list[int]:
        if not node: return [0, 0]
        
        left = traversal(node.left)
        right = traversal(node.right)
        
        # 偷当前节点：左孩子不偷 + 右孩子不偷 + 当前节点值
        val1 = node.val + left[0] + right[0]
        # 不偷当前节点：左右孩子偷不偷都行，取最大的
        val2 = max(left[0], left[1]) + max(right[0], right[1])
        
        return [val2, val1]
        
    res = traversal(root)
    return max(res[0], res[1])
```

### 4. 股票问题 (Stock Trading)

股票问题的神级模板：状态机 DP。统一使用二维数组维护当天的“持有”和“不持有”状态。

#### 买卖股票通用模板 (买卖多次，例：122.买卖股票的最佳时机II)

```python
def maxProfit(prices: list[int]) -> int:
    n = len(prices)
    if n == 0: return 0
    
    # dp[i][0]: 第 i 天【持有】股票的最大现金
    # dp[i][1]: 第 i 天【不持有】股票的最大现金
    dp = [[0] * 2 for _ in range(n)]
    
    # 初始化
    dp[0][0] = -prices[0]
    dp[0][1] = 0
    
    for i in range(1, n):
        # 持有两种可能：昨天就持有，或者今天刚买入（注意买卖次数限制决定买入时的公式）
        # 如果只能买一次 (121题): dp[i][0] = max(dp[i-1][0], -prices[i]) 
        # 如果可买多次 (122题): dp[i][0] = max(dp[i-1][0], dp[i-1][1] - prices[i])
        dp[i][0] = max(dp[i - 1][0], dp[i - 1][1] - prices[i]) 
        
        # 不持有两种可能：昨天就不持有，或者今天刚卖出
        dp[i][1] = max(dp[i - 1][1], dp[i - 1][0] + prices[i])
        
    return dp[-1][1]
```

### 5. 子序列问题 (Subsequence DP)

#### 最长递增子序列 (不连续 - LIS, 300.最长递增子序列)

```python
def lengthOfLIS(nums: list[int]) -> int:
    if not nums: return 0
    # dp[i]：以 nums[i] 结尾的最长递增子序列的长度
    dp = [1] * len(nums)
    result = 1
    
    for i in range(1, len(nums)):
        for j in range(i):
            if nums[i] > nums[j]:
                dp[i] = max(dp[i], dp[j] + 1)
        result = max(result, dp[i])
        
    return result
```

#### 最长公共子序列 / 最长重复子数组 (连续与不连续)

通用的二维网格模板，区别在于元素不相等时的处理。

```python
def longestCommonSubsequence(text1: str, text2: str) -> int:
    # dp[i][j]：text1 的前 i 个字符和 text2 的前 j 个字符的最长公共子序列长度
    dp = [[0] * (len(text2) + 1) for _ in range(len(text1) + 1)]
    
    for i in range(1, len(text1) + 1):
        for j in range(1, len(text2) + 1):
            if text1[i - 1] == text2[j - 1]:
                # 如果字符相同，在前一个状态上加 1
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                # 如果是求【不连续】公共子序列 (1143题)，则看前一个状态的最大值：
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
                # 如果是求【连续】重复子数组 (718题)，则不相等时直接为0，不需要这行else
                
    return dp[len(text1)][len(text2)]
```

#### 编辑距离问题 (Edit Distance, 72.编辑距离)

核心是处理“删除”、“替换”、“插入”操作对应的状态转移。

```python
def minDistance(word1: str, word2: str) -> int:
    # dp[i][j]: 以 i-1 结尾的 word1 转换为以 j-1 结尾的 word2 所需的最少操作数
    m, n = len(word1), len(word2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    # 初始化
    for i in range(m + 1): dp[i][0] = i
    for j in range(n + 1): dp[0][j] = j
        
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if word1[i - 1] == word2[j - 1]:
                # 字符相同，不需要操作
                dp[i][j] = dp[i - 1][j - 1]
            else:
                # 字符不同，取替换、删除/插入中的最小值 + 1
                dp[i][j] = min(
                    dp[i - 1][j - 1], # 替换
                    dp[i - 1][j],     # 删除 word1 的元素
                    dp[i][j - 1]      # 删除 word2 的元素 (等价于往 word1 插入)
                ) + 1
                
    return dp[m][n]
```

#### 回文串问题 (Palindrome, 516.最长回文子序列)

回文串问题由于要判断首尾，遍历顺序有特殊要求：**从下到上，从左到右**。

```python
def longestPalindromeSubseq(s: str) -> int:
    n = len(s)
    # dp[i][j]：字符串 s 在 [i, j] 范围内的最长回文子序列长度
    dp = [[0] * n for _ in range(n)]
    
    # 初始化，单个字符长度为1
    for i in range(n):
        dp[i][i] = 1
        
    # 注意遍历顺序：从下往上，从左往右
    for i in range(n - 1, -1, -1):
        for j in range(i + 1, n):
            if s[i] == s[j]:
                dp[i][j] = dp[i + 1][j - 1] + 2
            else:
                # 不相等时，加入左边或右边元素看哪个组成的回文长
                dp[i][j] = max(dp[i + 1][j], dp[i][j - 1])
                
    return dp[0][n - 1]
```

### 前缀和系列 (Prefix Sum)

前缀和主要用于快速求出数组中某一段连续区域的和，能将 $O(N)$ 的查询时间优化到 $O(1)$。核心思想是预先计算并保存前面元素的累加结果。

#### 一维前缀和 (1D Prefix Sum)

**场景：** 频繁查询一维数组 `nums` 中索引 `[left, right]` 范围内的元素之和。（例：303. 区域和检索 - 数组不可变）

*技巧：通常将前缀和数组的长度设为 `n + 1`，这样可以避免处理 `left=0` 时的越界判断问题。*

```python
class NumArray:
    def __init__(self, nums: list[int]):
        n = len(nums)
        # preSum[i] 表示 nums 中前 i 个元素的和 (即 nums[0] 到 nums[i-1])
        self.preSum = [0] * (n + 1)
        
        # 构建前缀和数组
        for i in range(1, n + 1):
            self.preSum[i] = self.preSum[i - 1] + nums[i - 1]

    def sumRange(self, left: int, right: int) -> int:
        # 查询 [left, right] 区间的和
        # 因为 preSum[right + 1] 包含了 nums[0...right] 的和
        # preSum[left] 包含了 nums[0...left-1] 的和
        # 两者相减即为 nums[left...right] 的和
        return self.preSum[right + 1] - self.preSum[left]
```

#### 二维前缀和 (2D Prefix Sum)

**场景：** 频繁查询二维矩阵 `matrix` 中左上角 `(row1, col1)` 到右下角 `(row2, col2)` 构成的子矩阵之和。（例：304. 二维区域和检索 - 矩阵不可变）

*技巧：基于容斥原理来计算当前前缀和以及求解目标子矩阵。*

```python
class NumMatrix:
    def __init__(self, matrix: list[list[int]]):
        if not matrix or not matrix[0]:
            return
            
        m, n = len(matrix), len(matrix[0])
        # preSum[i][j] 表示从 matrix[0][0] 到 matrix[i-1][j-1] 的子矩阵之和
        self.preSum = [[0] * (n + 1) for _ in range(m + 1)]
        
        # 构建二维前缀和数组
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                # 递推公式：当前位置的值 = 上方矩阵和 + 左方矩阵和 - 左上角重复计算的矩阵和 + 当前元素
                self.preSum[i][j] = (self.preSum[i - 1][j] 
                                   + self.preSum[i][j - 1] 
                                   - self.preSum[i - 1][j - 1] 
                                   + matrix[i - 1][j - 1])

    def sumRegion(self, row1: int, col1: int, row2: int, col2: int) -> int:
        # 查询目标子矩阵之和，同样利用容斥原理
        # 整体矩阵和 - 上方不在范围的矩阵 - 左侧不在范围的矩阵 + 左上角被多减去一次的矩阵
        return (self.preSum[row2 + 1][col2 + 1] 
              - self.preSum[row1][col2 + 1] 
              - self.preSum[row2 + 1][col1] 
              + self.preSum[row1][col1])
```

#### 一维差分数组 (1D Difference Array)

**场景：** 频繁对一维数组的某个区间 `[left, right]` 内的所有元素统一加上或减去一个值 `val`，最后求修改后的完整数组。（例：1094. 拼车、1109. 航班预订统计）

*技巧：差分数组是前缀和的逆运算。*

```python
class Difference:
    def __init__(self, nums: list[int]):
        n = len(nums)
        # diff[i] 记录 nums[i] 与 nums[i-1] 的差值
        self.diff = [0] * n
        self.diff[0] = nums[0]
        for i in range(1, n):
            self.diff[i] = nums[i] - nums[i - 1]
            
    def increment(self, left: int, right: int, val: int) -> None:
        # 对区间 [left, right] 增加 val
        self.diff[left] += val
        # 如果 right + 1 还没越界，说明 [right+1, 末尾] 的元素相对 [right] 并没有增加 val，所以要减去
        if right + 1 < len(self.diff):
            self.diff[right + 1] -= val
            
    def result(self) -> list[int]:
        # 根据差分数组反推（即求差分数组的前缀和），得到最终的修改后数组
        res = [0] * len(self.diff)
        res[0] = self.diff[0]
        for i in range(1, len(self.diff)):
            res[i] = res[i - 1] + self.diff[i]
        return res
```
