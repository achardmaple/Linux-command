### 二分查找法 

```python
from typing import List

class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        n = len(nums)
        left = 0
        right = n # 我们定义target在左闭右开的区间里，[left, right)  
        while left < right: # 因为left == right的时候，在[left, right)是无效的空间
            middle = left + ((right - left) >> 1)
            if nums[middle] > target:
                right = middle # target 在左区间，因为是左闭右开的区间，nums[middle]一定不是我们的目标值，所以right = middle，在[left, middle)中继续寻找目标值
            elif nums[middle] < target:
                left = middle + 1 # target 在右区间，在 [middle+1, right)中
            else: # nums[middle] == target
                return middle # 数组中找到目标值的情况，直接返回下标
        return right
```

### KMP

```python
def kmp(next_arr: List[int], s: str) -> None:
    next_arr[0] = -1
    j = -1
    for i in range(1, len(s)):
        while j >= 0 and s[i] != s[j + 1]:
            j = next_arr[j]
        if s[i] == s[j + 1]:
            j += 1
        next_arr[i] = j
```

def preorderTraversal(cur: TreeNode, vec: List[int]) -> None:
    if not cur:
        return
    vec.append(cur.val)                # 中 ，同时也是处理节点逻辑的地方
    preorderTraversal(cur.left, vec)   # 左

#### 深度优先遍历（迭代法）

前序遍历（中左右）
```python
def preorderTraversal(root: TreeNode) -> List[int]:
    result = []
    st = []
    if root:
        st.append(root)
    while st:
        node = st[-1]
        if node:
            st.pop()
            if node.right:
                st.append(node.right)  # 右
            if node.left:
                st.append(node.left)   # 左
            st.append(node)            # 中
            st.append(None)                          
        else:
            st.pop()
            node = st.pop()
            result.append(node.val)    # 节点处理逻辑
    return result
```

#### 广度优先遍历（队列）

```python
from collections import deque

def levelOrder(root: TreeNode) -> List[List[int]]:
    result = []
    if not root:
        return result
    que = deque([root])
    
    while que:
        size = len(que)
        vec = []
        for _ in range(size): # 这里一定要使用固定大小size，不要使用 len(que)
            node = que.popleft()
            vec.append(node.val)   # 节点处理的逻辑
            if node.left:
                que.append(node.left)
            if node.right:
                que.append(node.right)
        result.append(vec)
    return result
```

#### 二叉树深度

```python
def getDepth(node: TreeNode) -> int:
    if not node:
        return 0
    return 1 + max(getDepth(node.left), getDepth(node.right))
```

#### 二叉树节点数量

```python
def countNodes(root: TreeNode) -> int:
    if not root:
        return 0
    return 1 + countNodes(root.left) + countNodes(root.right)
```

### 并查集 

```python
n = 1005 # 根据题意而定 
father = [0] * n

# 并查集初始化
def init():
    for i in range(n):
        father[i] = i

# 并查集里寻根的过程
def find(u: int) -> int:
    if u == father[u]:
        return u
    father[u] = find(father[u]) # 路径压缩
    return father[u]

# 将v->u 这条边加入并查集
def join(u: int, v: int) -> None:
    u = find(u)
    v = find(v)
    if u == v:
        return 
    father[v] = u

# 判断 u 和 v是否找到同一个根
def same(u: int, v: int) -> bool:
    u = find(u)
    v = find(v)
    return u == v
```
